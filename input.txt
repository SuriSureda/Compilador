function printSum(nA : int, nB : int) : void {
	print(nA);
	print(" + ");
	print(nB);
	print(" = ");
	print((nA+nB));
	print("\n");
}

function main() {
	print("This program will print the sum 2 numbers : \n");
	print("Please enter first number : ");
	nA : int = read();
	print("Please enter second number : ");
	nB : int = read();

	printSum(nA, nB);
}
